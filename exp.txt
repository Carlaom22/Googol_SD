        ├── SearchGateway.java          # Interface RMI da Gateway
        ├── SearchGatewayImpl.java      # Gateway: decide qual barrel usar

        ├── CentralURLQueue.java        # Interface RMI da fila de URLs

        └── CentralURLQueueServer.java  # Publica a fila no RMI
